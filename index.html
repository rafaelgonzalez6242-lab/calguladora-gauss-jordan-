Algoritmo GaussJordan
    
    Definir x1y1, x2y1, x3y1 Como Real
    Definir x1y2, x2y2, x3y2 Como Real
    Definir x1y3, x2y3, x3y3 Como Real
    Definir r1, r2, r3 Como Real
    
    Escribir "RESOLUCION POR REDUCCION DE RENGLONES (GAUSS-JORDAN)"
    
    Escribir "Numero en la posicion 1,1: "
    Leer x1y1

    Leer x1y3
    Escribir "Numero en la posicion 2,1: "
    Leer x2y1
    Escribir "Numero en la posicion 2,2: "
    Leer x2y2
    Escribir "Numero en la posicion 2,3: "
    Leer x2y3
    Escribir "Numero en la posicion 3,1: "
    Leer x3y1
    Escribir "Numero en la posicion 3,2: "
    Leer x3y2
    Escribir "Numero en la posicion 3,3: "
    Leer x3y3

    Leer r1

    Leer r2

    Leer r3
    
    // PRIMER PIVOTE (1,1)//
    Si x1y1 <> 1 Entonces
        x1y2 <- x1y2 / x1y1
        x1y3 <- x1y3 / x1y1
        r1 <- r1 / x1y1
        x1y1 <- x1y1 / x1y1
    FinSi
    
    x2y2 <- x2y2 - (x2y1 * x1y2)
    x2y3 <- x2y3 - (x2y1 * x1y3)
    r2 <- r2 - (x2y1 * r1)
    x2y1 <- x2y1 - (x2y1 * x1y1)
    
    x3y2 <- x3y2 - (x3y1 * x1y2)
    x3y3 <- x3y3 - (x3y1 * x1y3)
    r3 <- r3 - (x3y1 * r1)
    x3y1 <- x3y1 - (x3y1 * x1y1)
    
    Escribir x1y1, " ", x1y2, " ", x1y3, " | ", r1
    Escribir x2y1, " ", x2y2, " ", x2y3, " | ", r2
    Escribir x3y1, " ", x3y2, " ", x3y3, " | ", r3
    Escribir ""
    
    // SEGUNDO PIVOTE (2,2)//
    Si x2y2 <> 1 Entonces
        x2y3 <- x2y3 / x2y2
        r2 <- r2 / x2y2
        x2y2 <- x2y2 / x2y2
    FinSi
    
    x1y3 <- x1y3 - (x1y2 * x2y3)
    r1 <- r1 - (x1y2 * r2)
    x1y2 <- x1y2 - (x1y2 * x2y2)
    
    x3y3 <- x3y3 - (x3y2 * x2y3)
    r3 <- r3 - (x3y2 * r2)
    x3y2 <- x3y2 - (x3y2 * x2y2)
    
    Escribir x1y1, " ", x1y2, " ", x1y3, " | ", r1
    Escribir x2y1, " ", x2y2, " ", x2y3, " | ", r2
    Escribir x3y1, " ", x3y2, " ", x3y3, " | ", r3
    Escribir ""
    
    // TERCER PIVOTE (3,3)//
    Si x3y3 <> 1 Entonces
        r3 <- r3 / x3y3
        x3y3 <- x3y3 / x3y3
    FinSi
    
    r1 <- r1 - (x1y3 * r3)
    x1y3 <- x1y3 - (x1y3 * x3y3)
    
    r2 <- r2 - (x2y3 * r3)
    x2y3 <- x2y3 - (x2y3 * x3y3)
    
    Escribir x1y1, " ", x1y2, " ", x1y3, " | ", r1
    Escribir x2y1, " ", x2y2, " ", x2y3, " | ", r2
    Escribir x3y1, " ", x3y2, " ", x3y3, " | ", r3
    Escribir ""
    
    Escribir "El resultado de x1: ", r1
    Escribir "El resultado de x2: ", r2
    Escribir "El resultado de x3: ", r3

Programa en C++



#include 
#include 
using namespace std;

int main() {
    float x1y1, x2y1, x3y1;
    float x1y2, x2y2, x3y2;
    float x1y3, x2y3, x3y3;
    float r1, r2, r3;

    cout << "===========================================================\n";
    cout << "  RESOLUCION DE SISTEMAS 3x3 POR GAUSS-JORDAN\n";
    cout << "===========================================================\n\n";

    cout << fixed << setprecision(4);

    // Entrada de coeficientes
    cout << "Coeficientes de la primera fila (x1 y1 y z1):\n";
    cin >> x1y1 >> x1y2 >> x1y3;

    cout << "Coeficientes de la segunda fila (x2 y2 y z2):\n";
    cin >> x2y1 >> x2y2 >> x2y3;

    cout << "Coeficientes de la tercera fila (x3 y3 y z3):\n";
    cin >> x3y1 >> x3y2 >> x3y3;

    cout << "Resultados de cada ecuacion (r1 r2 r3):\n";
    cin >> r1 >> r2 >> r3;

    // Primer pivote (1,1)
    if (x1y1 != 1) {
        x1y2 /= x1y1;
        x1y3 /= x1y1;
        r1    /= x1y1;
        x1y1  /= x1y1;
    }

    // Reducir columna 1 en fila 2
    x2y2 -= (x2y1 * x1y2);
    x2y3 -= (x2y1 * x1y3);
    r2   -= (x2y1 * r1);
    x2y1 -= (x2y1 * x1y1);

    // Reducir columna 1 en fila 3
    x3y2 -= (x3y1 * x1y2);
    x3y3 -= (x3y1 * x1y3);
    r3   -= (x3y1 * r1);
    x3y1 -= (x3y1 * x1y1);

    // Segundo pivote (2,2)
    if (x2y2 != 1) {
        x2y3 /= x2y2;
        r2   /= x2y2;
        x2y2 /= x2y2;
    }

    // Reducir columna 2 en fila 1
    x1y3 -= (x1y2 * x2y3);
    r1   -= (x1y2 * r2);
    x1y2 -= (x1y2 * x2y2);

    // Reducir columna 2 en fila 3
    x3y3 -= (x3y2 * x2y3);
    r3   -= (x3y2 * r2);
    x3y2 -= (x3y2 * x2y2);

    // Tercer pivote (3,3)
    if (x3y3 != 1) {
        r3   /= x3y3;
        x3y3 /= x3y3;
    }

    // Reducir columna 3 en fila 1
    r1   -= (x1y3 * r3);
    x1y3 -= (x1y3 * x3y3);

    // Reducir columna 3 en fila 2
    r2   -= (x2y3 * r3);
    x2y3 -= (x2y3 * x3y3);

    // Matriz final
    cout << "\nMatriz reducida final:\n";
    cout << x1y1 << "  " << x1y2 << "  " << x1y3 << "  |  " << r1 << endl;
    cout << x2y1 << "  " << x2y2 << "  " << x2y3 << "  |  " << r2 << endl;
    cout << x3y1 << "  " << x3y2 << "  " << x3y3 << "  |  " << r3 << endl << endl;

    // Solución final
    cout << "Soluciones:\n";
    cout << "x1 = " << r1 << endl;
    cout << "x2 = " << r2 << endl;
    cout << "x3 = " << r3 << endl;

    return 0;
}
